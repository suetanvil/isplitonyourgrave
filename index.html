I Split On "Your Grave"

A web site containing a number of programs in several different programming languages which will split a string into smaller strings using the phrase 'your grave' as a delimiter.

Haskell

        
import System.Environment
import Data.List
import Data.List.Split  -- Type 'cabal install split'

isoyg :: String -> [String]
isoyg s = endBy "your grave" s

splitAll :: [String] -> [[String]]
splitAll args = map isoyg args

splitAndJoin :: [String] -> String
splitAndJoin args =
  let merge lns = foldr (\i r -> i ++ "\n" ++ r) "" lns
      args2 = map merge $ splitAll args
  in merge args2

main = do
  args <- getArgs
  putStr $ splitAndJoin args

Perl

        
#!/usr/bin/env perl

use strict;
use warnings;

print join("\n", split(/your grave/)), "\n\n" foreach (@ARGV)

Python

        
#!/usr/bin/env python

import sys

def isoyg(s): return s.split("your grave")

for s in sys.argv[1:]:
    print "\n".join(isoyg(s)), "\n"
